% Paketdefinitionen für deutsche LaTeX Dokumente

% Pakete um deutsche Texte in LaTeX zu setzen:

% Wir setzen einen Text in neuer deutscher
% Rechtschreibung:
\usepackage[ngerman]{babel}
% Eingabetext ist im utf-8 Format:
\usepackage[utf8]{inputenc}
% Korrektes Ausgabeencoding für deutsche
% Umlaute:
\usepackage[T1]{fontenc}
% Korrekte Anführungsstriche in deutschen Texten:
\usepackage{csquotes}

% Pakete für den Mathematikmodus:
\usepackage{amsmath}
\usepackage{amssymb}
% Zum Setzen von Einheiten
\usepackage{siunitx}
% Wir wollen Zahlen/Einheiten nach
% deutschen Regeln setzen. siunitx nimmt
% leider nicht direkt die babel Spracheinstellung
\sisetup{locale = DE}

% Figuren speichern wir immer in einem Unterordner
% 'figuren' unter unseren LaTeX Dokumenten
\usepackage{graphicx}
\graphicspath{{figuren/}}

% Das Paket biblatex ist für Literaturverweise
% zuständig.
\usepackage[backend=biber,
            style=alphabetic % alphabetischer Zitierstil
           ]{biblatex}

% Meine Literaturdatenbank:
\addbibresource{literatur.bib}

% Sonstiges:

% Das Paket xspace sorgt für 'intelligente'
% Leerzeichnsetzung nach einem LaTeX Makro.
% siehe auch Datei my_newcommands_german.tex
\usepackage{xspace}

% Hin und wieder benötige ich ein wenig Dummy
% Text zum Vorführen wie hier in den Videos:
\usepackage{blindtext}

% Für das Einfügen von Web-Adressen:
\usepackage{url}

% Zum Erstellen von hyperlinks innerhalb des PDF
% Ausgabedokuments.
%
% Man beachte:
% - hyperref muss nahe am Ende aller Paketdefinitionen
%   stehen. Befragen Sie ggf. die Dokumentation
%   von hyperref, um zu sehen, welche Pakete 'nach'
%   hyoerref stehen müßen!
% - Löschen Sie alle alten LaTeX-Hilfsdateien, bevor
%   Sie ein Dokument mit hyperref übersetzen!
\usepackage{hyperref}
\hypersetup{
  pdfstartview=FitH,        % PDF Dokument so in den PDF Betrachter
                            % laden, dass das PDF-Betrachter Fenster
                            % horizontal ausgefüllt wird
  pdfauthor={Thomas Erben}, % Autor des PDF-Dokuments (für PDF Metadaten)
  pdftitle={Unser Dokument} % Titel des PDF-Dokuemnts (für PDF Metadaten)
}

% Um hyperref Links von Gleitobjekten an eine
% 'vernünftige' Stelle springen zu lassen:
\usepackage[all]{hypcap}
